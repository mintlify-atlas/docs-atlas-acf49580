This tutorial walks through creating an **incoming webhook integration** that receives HTTP requests from a third-party service and posts formatted messages to Zulip.

We'll use the [Hello World webhook](https://zulip.com/integrations/helloworld) as our example.

## What is a Webhook Integration?

A webhook integration:

1. Third-party service sends HTTP POST to Zulip
2. Zulip processes the payload
3. Zulip posts formatted message to a channel or DM

Examples: GitHub, Sentry, Jira, Stripe

## Prerequisites

- [Zulip development environment](/development/setup) set up
- Basic Python knowledge
- Understanding of HTTP and JSON
- Sample webhook payloads from the service

## Step 2: Write the Webhook Handler

**`zerver/webhooks/helloworld/view.py`**:

```python
from django.http import HttpRequest, HttpResponse

from zerver.decorator import webhook_view
from zerver.lib.response import json_success
from zerver.lib.typed_endpoint import JsonBodyPayload, typed_endpoint
from zerver.lib.validator import WildValue, check_string
from zerver.lib.webhooks.common import check_send_webhook_message
from zerver.models import UserProfile


@webhook_view("HelloWorld")
@typed_endpoint
def api_helloworld_webhook(
    request: HttpRequest,
    user_profile: UserProfile,
    *,
    payload: JsonBodyPayload[WildValue],
) -> HttpResponse:
    # Extract data from payload
    featured_title = payload["featured_title"].tame(check_string)
    featured_url = payload["featured_url"].tame(check_string)
    
    # Construct message
    body = "Hello! I am happy to be here! :smile:"
    body += f"\nThe Wikipedia featured article for today is **[{featured_title}]({featured_url})**"
    
    # Set topic
    topic_name = "Hello World"
    
    # Send message
    check_send_webhook_message(
        request,
        user_profile,
        topic_name,
        body
    )
    
    return json_success(request)
```

### Key Components

#### Decorators

```python
@webhook_view("HelloWorld")  # Integration name for analytics
@typed_endpoint              # Type-safe parameter handling
```

Use integration's official name in camel case.

#### Function Signature

```python
def api_helloworld_webhook(
    request: HttpRequest,      # Required
    user_profile: UserProfile, # Required
    *,
    payload: JsonBodyPayload[WildValue],  # JSON body
) -> HttpResponse:
```

Naming: `api_{integration_name}_webhook`

#### Payload Extraction

```python
# Safe extraction with validation
featured_title = payload["featured_title"].tame(check_string)

# With default
optional_field = payload.get("field", "default").tame(check_string)

# Nested data
user_name = payload["user"]["name"].tame(check_string)

# Arrays
commits = payload["commits"].tame(check_list(check_dict))
for commit in commits:
    message = commit["message"].tame(check_string)
```

#### Message Formatting

Use [Zulip Markdown](/api/message-formatting):

```python
body = f"""**Title**: {data['title']}

{data['description']}

* Item 1
* Item 2

[Link]({data['url']}) :tada:
"""
```

#### Sending Messages

```python
check_send_webhook_message(
    request,
    user_profile,
    topic_name,  # Topic for channel messages
    body,        # Message content
    unquote_url_parameters=True  # Optional: decode URL params
)
```

This automatically:
- Posts to channel if `stream` param in URL
- DMs bot owner otherwise
- Returns proper HTTP responses

    "github",

### Fixtures with HTTP Headers

Some services use headers for event type:

```python
# In view.py
def api_github_webhook(
    request: HttpRequest,
    user_profile: UserProfile,
    *,
    payload: JsonBodyPayload[WildValue],
) -> HttpResponse:
    event = get_event_header(request, "X-GitHub-Event", "GitHub")
    
    if event == "push":
        return handle_push(payload, ...)
    elif event == "issues":
        return handle_issue(payload, ...)
```

Document in tests:

```python
def get_http_headers(self, name: str) -> dict[str, str]:
    if name == "push":
        return {"X-GitHub-Event": "push"}
    return {}
```

        expected_message = (
            "Hello! I am happy to be here! :smile:\n"
            "The Wikipedia featured article for today is "
            "**[Marilyn Monroe](https://en.wikipedia.org/wiki/Marilyn_Monroe)**"
        )

            "hello",

            expected_message,

## Advanced Topics

### Multiple Event Types

Handle different events:

```python
@webhook_view("MyService")
@typed_endpoint
def api_myservice_webhook(
    request: HttpRequest,
    user_profile: UserProfile,
    *,
    payload: JsonBodyPayload[WildValue],
) -> HttpResponse:
    event_type = payload["event"].tame(check_string)
    
    if event_type == "issue.created":
        return handle_issue_created(request, user_profile, payload)
    elif event_type == "issue.updated":
        return handle_issue_updated(request, user_profile, payload)
    else:
        raise UnsupportedWebhookEventTypeError(event_type)

def handle_issue_created(
    request: HttpRequest,
    user_profile: UserProfile,
    payload: WildValue,
) -> HttpResponse:
    # Extract data
    issue = payload["issue"]
    title = issue["title"].tame(check_string)
    
    # Format message
    topic = f"Issue #{issue['id'].tame(check_int)}"
    body = f"New issue: **{title}**"
    
    # Send
    check_send_webhook_message(request, user_profile, topic, body)
    return json_success(request)
```

### URL Parameters

Custom parameters in webhook URL:

```python
@typed_endpoint
def api_github_webhook(
    request: HttpRequest,
    user_profile: UserProfile,
    *,
    payload: JsonBodyPayload[WildValue],
    branches: str | None = None,  # ?branches=main,dev
    only_issues: bool = False,    # ?only_issues=true
) -> HttpResponse:
    # Filter by branches
    if branches:
        branch_list = branches.split(",")
        if payload["ref"].tame(check_string) not in branch_list:
            return json_success(request)  # Skip this event
    
    # Skip if not an issue
    if only_issues and payload["event"] != "issues":
        return json_success(request)
    
    # Process...
```

### Form Data (Not JSON)

Some services send URL-encoded data:

```python
from zerver.lib.typed_endpoint import typed_endpoint_without_parameters

@webhook_view("FormService")
@typed_endpoint_without_parameters
def api_formservice_webhook(
    request: HttpRequest,
    user_profile: UserProfile,
) -> HttpResponse:
    # Access POST data
    title = request.POST.get("title", "")
    description = request.POST.get("description", "")
    
    body = f"**{title}**\n\n{description}"
    check_send_webhook_message(request, user_profile, "Form", body)
    return json_success(request)
```

### Ignored Events

Silently ignore certain events:

```python
IGNORED_EVENTS = {"ping", "status", "heartbeat"}

def api_myservice_webhook(...) -> HttpResponse:
    event = payload["event"].tame(check_string)
    
    if event in IGNORED_EVENTS:
        return json_success(request)  # Success, but no message
    
    # Process other events...
```

- ✓ One fixture per event type
- ✓ Test both success and error cases
